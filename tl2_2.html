Décomposition du signal avec la transformée en ondelettes discrètes

wname = 'sym20';
level = 10;
[spectre,info]=wavedec(z,level,wname );
d = detcoef(spectre, info, 1:1:level);
t= [];

Traitement

%Energie des coefficients
for i=1:1:level
Sum=0;
Sum = sum(d{i}.^2, 'all');
t=[t Sum];
end
%Coefficients avec le plus d'énergie
[C,I]=maxk(t,2);
%Coupe du signal
index_separation=floor((I(1)+I(2))/2);
size_d=[];
for i=1:1:level
size_d=[size_d length(d{i})];
end
size_x=0;
for i=1:1:index_separation
size_x=size_x+ info(end-i) ;
end
